```python
def dilate_mask_by_mm(mask: np.ndarray, spacing_xyz: tuple[float, float, float], radius_mm: float) -> np.ndarray:
    spacing_zyx = spacing_xyz[::-1]
    radius_voxels = [max(1, int(np.ceil(radius_mm / spacing))) for spacing in spacing_zyx]
    grid = np.ogrid[tuple(slice(-r, r + 1) for r in radius_voxels)]
    distance = sum((axis * spacing) ** 2 for axis, spacing in zip(grid, spacing_zyx, strict=True))
    structure = distance <= radius_mm**2
    return ndi.binary_dilation(np.asarray(mask, dtype=bool), structure=structure)
```

Add `apply_exclusion_seed_override(exclusion, seed)` returning `exclusion & ~seed`.
